        Leer nombre;
        Escribir "Tambien sus apellidos";
        Leer apellidos;
        Escribir "Ingrese su edad";
        Leer edad;
        
        Escribir nombre, " ", apellidos;

            Escribir "Usted es mayor de edad, por lo tanto puede entrar

            Escribir "Usted es menor de edad, por lo tanto, no puede 
                entrar a la fiesta, por favor devuélvase a su casa.";
        FinSi
    FinProceso
